10. 转义字符

print("#" * 75) #####

tabby_cat = "\tI'm tabbed in."           # <\t> : tab
persian_cat = "I'm split\non a line."    # <\n> : escape lines
backslash_cat = "I'm \\ a \\ cat."       # <\\> : \

fat_cat = """                            
I'll do a list:
\t* cat food
\t* fishied
\t* catnip\n\t* grass """                # <"""> : code multiple lines

print(tabby_cat)
print(persian_cat)
print(backslash_cat)
print(fat_cat)
