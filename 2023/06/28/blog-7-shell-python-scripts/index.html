1.使用python作图的脚本(一)

#!/usr/bin/python3 python3
import matplotlib.pyplot as plt
import numpy as np

x,y = np.loadtxt('input.dat', delimiter = ',', usecols=(0, 1), unpack=True)

plt.xlabel('x / units')
plt.ylabel('y / units')
plt.plot(x,y, 'rs-', linewidth=2.0)
plt.show()

2.使用python作图的脚本(二)

#!/usr/bin/python3 python3
import matplotlib.pyplot as plt
import numpy as np

x,y,z = np.loadtxt('input.dat', delimiter = ',', usecols=(0, 1, 2), unpack=True)

plt.xlabel('x / s')
plt.ylabel('y / s')
plt.plot(x,y, 'rs-', linewidth=2.0)
plt.plot(x,z*4, 'bs-', linewidth=2.0)
plt.show()

# 3.二维两列数据作图的脚本(一)

```bash
#!/usr/bin/env bash
echo "This script is used to plot 2D diagram"

function_input()
{
echo -n "Please name the plot script: (.py)"
read name
echo -n "Please enter the plot data file: "
read data
echo -n "Please name the xlabel(' / '): "
read xlabel
echo -n "Please name the ylabel(' / '): "
read ylabel
}

function_input $name $data $xlabel $ylabel

cat > $name <<!
#!/usr/bin/env python3
import matplotlib.pyplot as plt
import numpy as np

x,y = np.loadtxt('$data', delimiter = ' ', usecols=(0, 1), unpack=True)
plt.xlabel('$xlabel')
plt.ylabel('$ylabel')
plt.plot(x,y, 'rs-', linewidth=2.0)
plt.show() 
!
python3 $name

4.二维三列数据作图的脚本(二)

#!/usr/bin/env bash
echo "This script is used to plot 3D diagram"
echo -n "Please name the plot script(.py): "
read name
echo -n "Please enter the plot data file: "
read data
echo -n "Please name the xlabel(' / '): "
read xlabel
echo -n "Please name the ylabel(' / '): "
read ylabel

cat > $name <<!
#!/usr/bin/env python3

import matplotlib.pyplot as plt
import numpy as np

x,y,z = np.loadtxt('$data', delimiter = ' ', usecols=(0, 1, 2), unpack=True)
plt.xlabel('$xlabel')
plt.ylabel('$ylabel')
plt.plot(x,y, 'rs-', linewidth=2.0)
plt.plot(x,z*4, 'bs-', linewidth=2.0)
plt.show() 
!

python3 $name
