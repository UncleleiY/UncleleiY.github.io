Recording of python study

21.1 how to use the key word: return , to send the value back

def add(a, b):
    print(f"add {a} + {b}")
    return a + b
    
def subtract(a, b):
    print(f"subtract {a} - {b}")
    return a - b
    
def multiply(a, b):
    print(f"multiply {a} * {b}")
    return a * b

def divide(a, b):
    print(f"divide {a} / {b}")
    return a / b
    
print("let's do some math with just functions !")

age = add(20, 3)
height = subtract(166, 0)
weight = multiply(45, 2)
IQ = divide(250, 2)

print(f"age: {age}, height: {height}, weight: {weight}, IQ: {IQ}")

amazing = add(age, subtract(height, multiply(weight, divide(IQ, 2))))

print("wasn't it amazing ?", amazing, "yeah, it's funny !")
