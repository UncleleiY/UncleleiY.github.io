//第二题，计算BMI；ok，所设置的变量应该有意义；

/*

{
	using namespace std;
	double feet, inches, pound, Eheight, Cheight, weight, BMI;
	const double h = 12;
	const double m = 0.0254;
	const double w = 2.2;
	cout << "enter your height with feet and inches :";
	cin >> feet >> inches;
	cout << "enter your weight with pounds :";
	cin >> pound;
	Eheight = feet * h + inches;
	Cheight = Eheight * m;
	weight = pound / w;
	BMI = weight / (Cheight * Cheight);

}

//第四题，时间转换；可以使用减法代替取余方式；

/*
#include<iostream>
int main(void)
{
	using namespace std;
	long second, days, hours, minutes, seconds;
	const long t = 60;
	const long k = 24;
	cout << "Enter the number of seconds : ";
	cin >> second;
	//days = second / (t*t*k);
	//hours = (second - days*t*t*k)/(t*t);
	//minutes = (second- days*t*t*k-hours*t*t)/t;
	//seconds = second - days * t * t * k - hours * t * t-minutes*t;
	days = second / (t * t * k);
	second = second % (t * t * k);
	hours = second / (t * t);
	second = second % (t * t);
	minutes = second / t;
	seconds = second % t;
	cout << second << " seconds = " << days << " days, " << hours << " hours, " << minutes << " minutes, " << seconds << " seconds ." << endl;
	return 0;
