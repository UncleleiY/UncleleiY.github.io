## how to define variables 
a = "hello, dear ~"
## how to print it 
print(f"{a}, my pleasure to meet you here !")
print('*'*50)

## how to write codes more than one line: using \
argv1 = 1 ; argv2 = 2 ; argv3 = 3 ; argv4 = 4 ;
b = argv1 + \
    argv2 + \
    argv3 + \
    argv4 
print(f"1 + 2 + 3 + 4 = {b}")
print('*'*50)
'''
here be attention the order of define variables and use them !!!
don't use the variable that you never define before !!! 
'''

## there are four kinds of number: int, bool, float, complex(a + bi)
int1 = 1
bool0 = False
bool00 = 'false'
float1 = 2E5
complex1 = 1 + 1j
print(f"""int1 = {int1}, 
bool0 = {bool0}, 
bool00 = {bool00}, 
float1 = {float1}, 
complex1 = {complex1}""")
print('*'*50)

## \ can be used to change initial meaning, but we add r in front of the string to avoid 
a = 'print("this is line with \\n")'
b = 'print(r"this is line with \\n")' 
print(f"{a}:")
print('this is line with \n')
print(f"{b}:")
print(r'this is line with \n')
print('*'*50)

## how to conject strings using + and *
a = 'this '
b = 'is '
c = 'a'
x = ' '
d = 'string'

print(f"{a}{b}{c}{x}{d}")
print(a + b + c + x + d)
print(a*2 + b*2 + c*2 + x + d*2)
print('*'*50)

## how to print strings in different ways
strs='1234567890'

print(strs)
print(strs[0:-1])
print(strs[2:5])
print(strs[2:])
print(strs[1:5:2])
print(strs * 3)
print(strs + 'love')
print(strs + '\\n')
print(r'\n')
print('\\n')
print('*' * 50)

## how to write more statements in one line using <;>
import sys; x = 'love never fade away'; sys.stdout.write(x + '\n')
print('*' * 50)
## how to print without shifting using end=" "
x="love"
y="never"
z="fade"
# shift
print(x)
print(y)
print(z)
# no shift
print(x, end=" ")
print(y, end=" ")
print(z, end=" ")
print()
print('*' * 50 + '\n')


# how to use 'import' to input some functions that you need
# here we are going to find the path of python3  

import sys
print('*' * 25 + " import module " + '*' * 25) 
print("the arguments list as following: ")
for i in sys.argv:
    print(i)
print('\n the path of python:',sys.path)

from sys import argv,path
print('*' * 25 + " import module " + '*' * 25)
print('path:',path)
