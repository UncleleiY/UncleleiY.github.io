shell-vasp 任务目录脚本

vasp生成任务目录的脚本

#!/usr/bin/env bash
echo """
This script is used to make some new directories 
and then execute different calculation-tasks for you choose
Now the following tasks can be executed: (scf) (dos) (band) 
"""

function_scf()
{
mkdir scf && tar czvf chg.tgz CHG* &&
cp INCAR KPOINTS POSCAR OUTCAR chg.tgz scf/.
}

function_dos()
{
mkdir dos &&  
cp INCAR KPOINTS POSCAR OUTCAR DOSCAR dos/.
}

function_band()
{
mkdir band &&  
cp INCAR KPOINTS POSCAR OUTCAR EIGENVAL band/.
}

echo -n "Please choose which kind of task you're going to do: "
read task

a=scf 
b=dos
c=band

if [ $a == $task ]
then
    function_scf
    echo "scf directory writing successfully !"
elif [ $b == $task ]
then 
    function_dos
    echo "dos directory writing successfully !"
elif [ $c == $task ]
then
    function_band
    echo "band directory writing successfully !"
fi
scripts
vasp shell
