#include<iostream>
using namespace std;//
const int SIZE = 10;
void dichotomy(int arr[], int len, int num)
{
	int front = 0, end = len, mid = 0;
	while (front < end)
	{
		mid = (front + end) / 2;
		if (arr[mid] > num)
		{
			end = mid - 1;
		}
		else if (arr[mid] < num)
		{
			front = mid + 1;
		}
		else
		{
			cout << num << " is the " << (mid + 1) << " element." << endl;
			break;
		}
	}
	if (front >= end)
	{
		cout << "There is no element!" << endl;
	}
}
int test(int arr[], int len, int num)
{
	int low = 0, high = len, mid = 0;
	while (low <= high)
	{
		mid = (low + high) / 2;
		if (arr[mid] == num)
		{
			return mid;
		}
		else if (arr[mid] > num)
		{
			high = mid - 1;
		}
		else
		{
			low = mid + 1;
		}
	}
	return NULL;
}
int main()
{
	int arr[SIZE] = { 1,3,5,7,9,11,13,17,19,23 };
	int len = sizeof(arr) / sizeof(arr[0]);
	int num;
	cout << "Please enter your num:";
	cin >> num;
	//	dichotomy(arr, len, num);
	cout << test(arr, len, num) << endl;

	return 0;
}
