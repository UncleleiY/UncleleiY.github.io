懒狗科研（二）

辅助脚本（by Uncle-yang）

自己写的一个小脚本，用于免除大量重复操作：

  • 结构优化后自洽的参数辅助修改
  • 自洽完成后能带态密度参数辅助修改以及后处理
  • 自洽完成后bader和ELF计算参数修改以及后处理
#！/bin/bash

# Attention ! Make sure you finished GeoOpt calculation and stay in the GeoOpt directory waiting for the next step !
#############################################<<<<scf-script>>>>#######################################################
scf(){
mkdir scf
cp INCAR KPOINTS POTCAR slurm.sh scf/
cp CONTCAR scf/POSCAR
cd scf/
sed -i '/\bISTART\b/s/.*/ISTART = 0/' INCAR
sed -i '/\bICHARG\b/s/.*/ICHARG = 2/' INCAR
sed -i '/\bLWAVE\b/s/.*/LWAVE = .TRUE./' INCAR
sed -i '/\bLCHARG\b/s/.*/LCHARG = .TRUE./' INCAR
sed -i '/\bIBRION\b/s/.*/IBRION = -1/' INCAR
sed -i '/\bNSW\b/s/.*/NSW = 0/' INCAR
cd ..
echo "Attention ! All arguments rely on your calculation system !"
echo "Attention ! Check the value of IBRION, NSW, ENCUT, ISMEAR, SIGMA, KPOINTS !"
}
#############################################<<<<scf-script>>>>#######################################################


#############################################<<<<band-script>>>>######################################################
band(){
cd scf/
mkdir band
cp INCAR POSCAR POTCAR WAVECAR CHGCAR slurm.sh band/
mv band/ ..
cd ../band/
sed -i '/\bISTART\b/s/.*/ISTART = 1/' INCAR
sed -i '/\bICHARG\b/s/.*/ICHARG = 11/' INCAR
sed -i '/\bLWAVE\b/s/.*/LWAVE = .FALSE./' INCAR
sed -i '/\bLCHARG\b/s/.*/LCHARG = .FALSE./' INCAR
sed -i '/\bISMEAR\b/s/.*/ISMEAR = 0/' INCAR
sed -i '/\bSIGMA\b/s/.*/SIGMA = 0.05/' INCAR
sed -i '/\bLORBIT\b/s/.*/LORBIT = 11/' INCAR
echo -e "3\n302\n" | vaspkit > information
cp KPATH.in KPOINTS
cd ..
echo "Attention ! Check the value of SIGMA, LORBIT, NBANDS !"
echo "Attention ! Use primitive-cell to do band calculation !"
echo "Attention ! Check the KPATH whether it meets your system !"
}

band_data_process(){
cd band/
echo -e "21\n211\n1\n"| vaspkit > information
echo -e "61\n"| yang
cd ..
}
#############################################<<<<band-script>>>>######################################################


#############################################<<<<dos-script>>>>######################################################
dos(){
cd scf/
mkdir dos
cp INCAR KPOINTS POSCAR POTCAR WAVECAR CHGCAR slurm.sh dos/
mv dos/ ..
cd ../dos/
sed -i '/\bISTART\b/s/.*/ISTART = 1/' INCAR
sed -i '/\bICHARG\b/s/.*/ICHARG = 11/' INCAR
sed -i '/\bLWAVE\b/s/.*/LWAVE = .FALSE./' INCAR
sed -i '/\bLCHARG\b/s/.*/LCHARG = .FALSE./' INCAR
sed -i '/\bISMEAR\b/s/.*/ISMEAR = -5/' INCAR
sed -i '/\bLORBIT\b/s/.*/LORBIT = 11/' INCAR
cd ..
echo "Attention ! Check the value of LORBIT, NEDOS, ISMEAR !"
echo "Attention ! All arguments rely on your calculation system !"
}

dos_data_process(){
cd dos/
echo -e "11\n111\n1\n"| vaspkit > information
echo -e "71\n"| yang 
cd ..
}
#############################################<<<<dos-script>>>>######################################################


#############################################<<<<ELF-script>>>>######################################################
ELF(){
cd scf/
mkdir ELF
cp POSCAR POTCAR WAVECAR slurm.sh ELF/
mv ELF/ ..
cd ../ELF/
echo -e "1\n101\nEL\n"| vaspkit > information
cd ..
}
#############################################<<<<ELF-script>>>>######################################################


#############################################<<<<bader-script>>>>####################################################
bader_charge(){
cd scf/
mkdir bader
cp POSCAR POTCAR slurm.sh bader/
mv bader/ ..
cd ../bader/
echo -e "1\n101\nBD\n"| vaspkit > information
cd ..
}

bader_data_process(){
cd bader/
chgsum.pl AECCAR0 AECCAR2 
wait
bader CHGCAR -ref CHGCAR_sum 
cd ..
}
#############################################<<<<bader-script>>>>#####################################################


#############################################<<<<auto-script>>>>#####################################################

#############################################<<<<control-script>>>>###################################################
echo "This script is used to help you to be a lazy dog !"
echo "Attention ! Make sure you finished GeoOpt calculation and stay in the GeoOpt directory !"
echo "Attention ! SCF calculation must be the first step before you start NONSCF calculation !"
echo """-----------------------------------------------------------------------------------------
1) scf                                               
2) band(2D)                                            
3) band-plot(simple view)                                  
4) dos(2D)                                                       
5) dos-plot(simple view)                                            
6) ELF                                                             
7) bader                                                             
8) bader_data_process                                
9)  auto_submit(band&&dos)                                            
10) auto_process(band&&dos)                                           
-----------------------------------------------------------------------------------------"""
echo "Here you can choose the tasks as shown above ----> "

read task
case $task in
    1) scf ;;
    2) band ;;
    3) band_data_process ;;
    4) dos ;;
    5) dos_data_process ;;
    6) ELF ;;
    7) bader_charge ;;
    8) bader_data_process ;;
    9) auto_submit ;;
    10) auto_process ;;
    *) echo "please check the inputation !" ;;
esac
#############################################<<<<control-script>>>>###################################################
