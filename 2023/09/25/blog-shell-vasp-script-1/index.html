懒狗科研（一）

辅助脚本（by Uncle-yang）

自己写的一个小脚本，用于免除大量重复操作：

  • 参数测试脚本生成及简单数据处理
  • 创建任务目录以及复制相关文件
  • 调度其他辅助脚本的操作主盘
#!/bin/bash

########################################<<<<slurm-script>>>>#####################################
f11(){
cat > slurm.sh <<EOF
#!/bin/bash
#SBATCH -J vasp                                 ## Job name
#SBATCH -N 1                                    ## Number of nodes
#SBATCH --ntasks-per-node=64                    ## Each node has 64 tasks
#SBATCH -p xahcnormal                           ## operation Partition
#SBATCH -t 10:00:00                             ## time for your job eg:02-02:02:02 means 2d2h2min2s
module purge
module load compiler/intel/2017.5.239
module load mpi/intelmpi/2017.4.239
export MKL_DEBUG_CPU_TYPE=5  
export MKL_CBWR=AVX2      
export I_MPI_PIN_DOMAIN=numa   
#srun --mpi=pmi2  /path/to/vasp_std
srun --mpi=pmi2 /work/home/yanglei1/software/vasp.5.4.4/bin/vasp_std
EOF
}
########################################<<<<slurm-script>>>>#####################################


########################################<<<<encut-script>>>>#####################################
f21(){
cat > encut.sh <<'EOF'
#!/bin/bash

num="300 350 400 450 500"

# remove the old dirs
for i in $num ; do rm -rf Encut$i; done

# make a set of dirs for encut test
for i in $num ; do
mkdir Encut$i && cd Encut$i
cp ../INCAR .
cp ../POTCAR .
cp ../POSCAR .
cp ../KPOINTS .
cp ../slurm.sh .
sed -i "s/^.*\\bENCUT\\b.*$/ENCUT = $i/" INCAR

sbatch slurm.sh 
cd ..
done
EOF
}

f22(){
rm encut.dat
for dir in Encut*; do
        Encut=$(grep "ENCUT" $dir/INCAR | tail -1 | awk '{print $3}')
        Energy=$(grep "  without" $dir/OUTCAR | tail -1 | awk '{printf "%12.6f\n", $4}')
        echo $Encut $Energy >> encut.dat
done

cat encut.dat

cat > encut.py <<EOF
#!/usr/bin/env python3
import matplotlib.pyplot as plt
import numpy as np
x,y = np.loadtxt('encut.dat', delimiter = ' ', usecols=(0, 1), unpack=True)
plt.xlabel('ENCUT/eV')
plt.ylabel('Total energy/eV')
plt.plot(x,y, 'rs-', linewidth=2.0)
plt.savefig('encut.png')
plt.show()
EOF
python3 encut.py
}

f32(){
rm sigma.dat
for dir in Sigma*; do
        r=$(grep "SIGMA" $dir/INCAR | tail -1 | awk '{print $3}')
        TS=$(grep "entropy T" $dir/OUTCAR | tail -1 | awk '{printf "%12.8f \n", $5}')
        echo $r $TS >> sigma.dat
done

cat sigma.dat

cat > sigma.py <<EOF
#!/usr/bin/env python3
import matplotlib.pyplot as plt

EOF
python3 sigma.py
}

f42(){
rm kpoints.dat
for dir in */ ; do

########################################<<<<director-script>>>>#####################################
f51(){
echo "This script is used to make some new directories !"
echo "Make sure to stay in the GeoOpt directory right now !"
echo """-----------------------------------------------------
1) scf
2) dos
3) band
4) ELF
5) bader
6) phonon
-----------------------------------------------------"""
echo "Here you can choose tasks as shown above ----> "
read inputs
case $inputs in 
        1)
        mkdir scf
        cp INCAR KPOINTS CONTCAR POTCAR slurm.sh scf/
        mv scf/CONTCAR scf/POSCAR
        echo "scf directory writing successfully !"
        ;;
        2)
        mkdir dos
        cp INCAR KPOINTS POSCAR POTCAR WAVECAR CHGCAR slurm.sh dos/
        echo "dos directory writing successfully !"
        ;;
        3)
        mkdir band
        cp INCAR POSCAR POTCAR WAVECAR CHGCAR slurm.sh band/
        echo "band directory writing successfully !"
        ;;
        4)
        mkdir ELF
        cp INCAR POSCAR POTCAR WAVECAR slurm.sh ELF/
        echo "ELF directory writing successfully !"
        ;;
        5)
        mkdir bader
        cp POSCAR POTCAR slurm.sh bader/
        echo "bader directory writing successfully !"
        ;;
        6)
        mkdir phonon
        cp CONTCAR POTCAR KPOINTS slurm.sh phonon/
        echo "phonon directory writing successfully !"
        ;;
esac
}

f61(){
#!/bin/bash
echo -e "check your directory : scf/ band/ dos/ , all of them are existing in the same directory GeoOpt/ \n "
# Fermi level of scf
fermi=`grep E-fermi ../scf/OUTCAR | awk '{printf "%12.6f \n", $3 }'`
fermi_number=`grep efermi ../scf/vasprun.xml -n | awk '{printf "%d\n", $1}'`
echo -e "fermi level of scf : \n $fermi_number $fermi"

# Fermi level of dos
cp ../dos/vasprun.xml dos_vasprun.xml
dosfermi1_number=`grep efermi dos_vasprun.xml -n | awk '{printf "%d\n", $1}'`
dosfermi1=`grep efermi dos_vasprun.xml | awk '{printf "%12.6f \n", $3 }'`
echo -e "fermi level of dos : \n $dosfermi1_number $dosfermi1"
sed -i "/\befermi\b/s/.*/ <i name=\"efermi\">      $fermi <\/i> /" dos_vasprun.xml
sleep 1s
dosfermi2=`grep efermi dos_vasprun.xml | awk '{printf "%12.6f \n", $3 }'`
echo "After modfied, fermi level of dos : $dosfermi2"

# Fermi level of band
cp vasprun.xml band_vasprun.xml
bandfermi1_number=`grep efermi band_vasprun.xml -n | awk '{printf "%d\n", $1}'`
bandfermi1=`grep efermi band_vasprun.xml | awk '{printf "%12.6f \n", $3 }'`
echo -e "fermi level of band : \n $bandfermi1_number $bandfermi1"
sed -i "/\befermi\b/s/.*/ <i name=\"efermi\">      $fermi <\/i> /" band_vasprun.xml
sleep 1s
bandfermi2=`grep efermi band_vasprun.xml | awk '{printf "%12.6f \n", $3 }'`
echo -e "After modfied, fermi level of band : $bandfermi2 \n"


# plot the banddos
cat > xxx_banddos_plot_py_xxx <<EOF
import matplotlib.pyplot as plt
from pymatgen.electronic_structure.plotter import (BSDOSPlotter, BSPlotter,
                                                   BSPlotterProjected,
                                                   DosPlotter)
from pymatgen.io.vasp.outputs import Vasprun
# read vasprun.xml，get band and dos information
bs_vasprun = Vasprun("band_vasprun.xml",parse_projected_eigen=True)
bs_data = bs_vasprun.get_band_structure(line_mode=True)
dos_vasprun=Vasprun("dos_vasprun.xml")
dos_data=dos_vasprun.complete_dos
# set figure parameters, draw figure
banddos_fig = BSDOSPlotter(bs_projection="elements",
                           dos_projection="elements",
                           vb_energy_range=4,
                           cb_energy_range=3,
                           font = 'Times New Roman',
                           axis_fontsize = 36,
                           tick_fontsize = 28,
                           legend_fontsize = 28,
                           fig_size = (30,20))
banddos_fig.get_plot(bs=bs_data, dos=dos_data)
plt.savefig('banddos.pdf')
plt.savefig('banddos.svg')
EOF
python3 xxx_banddos_plot_py_xxx
}

########################################<<<<dosplot-script>>>>######################################
f71(){
cat > dosplot.py <<EOF
#!/bin/env bash
import matplotlib.pyplot as plt
from pymatgen.io.vasp.outputs import Vasprun
from pymatgen.electronic_structure.plotter import BSDOSPlotter,BSPlotter,BSPlotterProjected,DosPlotter

# read vasprun.xml and get band and dos information
bs_vasprun=Vasprun("./vasprun.xml",parse_projected_eigen=True)
bs_data=bs_vasprun.get_band_structure(line_mode=False)
dos_vasprun=Vasprun("./vasprun.xml")
dos_data=dos_vasprun.complete_dos

# set figure parameters, draw figure
plt_dos = DosPlotter(stack=False, sigma=0.5) # sigma is changable
plt_dos.add_dos('total dos',dos=dos_data)
plt_dos.add_dos_dict(dos_data.get_spd_dos())
plt_dos.save_plot('dosspd.pdf',img_format='pdf')
EOF
python3 dosplot.py
}
